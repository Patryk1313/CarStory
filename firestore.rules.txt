rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    function isSignedIn() {
      return request.auth != null;
    }

    function isOwnUser(userId) {
      return isSignedIn() && request.auth.uid == userId;
    }

    function isLegacyHistoryEntry() {
      return !resource.data.keys().hasAny(["ownerUid"]);
    }

    function isHistoryOwner() {
      return isSignedIn() && (
        isLegacyHistoryEntry() ||
        resource.data.ownerUid == request.auth.uid
      );
    }

    function isWritingOwnHistory() {
      return isSignedIn() &&
        request.resource.data.ownerUid == request.auth.uid;
    }

    match /users/{userId} {
      allow read, write: if isOwnUser(userId);

      match /vehicles/{vehicleId} {
        allow read, write: if isOwnUser(userId);

        match /serviceHistory/{entryId} {
          allow read, write: if isOwnUser(userId);
        }
      }
    }

    // Legacy paths kept temporarily so the app can migrate existing data.
    match /cars/{carId}/serviceHistory/{entryId} {
      allow create: if isWritingOwnHistory();
      allow read, delete: if isHistoryOwner();
      allow update: if isHistoryOwner() && isWritingOwnHistory();
    }

    match /cars/{carId}/profiles/{userId} {
      allow read, write: if isOwnUser(userId);
    }
  }
}
